import { existsSync }  from "fs-extra"
import { resolve } from "path"
import { exec } from "child_process"

const WORKFLOW_NAME = "Open NG-Xterm here.workflow"
const INSTALL_PATH = resolve(process.env.HOME!, "Library", "Services")
const RESOURCE_PATH = resolve(process.cwd(), "lib", "integrations", "resources", WORKFLOW_NAME)
const INSTALLED_PATH = resolve(process.env.HOME!, "Library", "Services", WORKFLOW_NAME)

export class MacIntegration {

	/**
	 * Check if the integration is installed, return true in this
	 * case and false else.
	 *
	 * @returns A promise of if the integration is installed
	 */
	async isInstalled(): Promise<boolean> {
		return existsSync(INSTALLED_PATH)
	}

	/**
	 * Install the integration by copying the workflow from the
	 * resources directory to the install path.
	 *
	 * @returns A promise when the installation has completed
	 */
	async installWorkFlow(): Promise<void> {
		exec(`cp -r "${ RESOURCE_PATH }" "${ INSTALL_PATH }"`, ((error, stdout) => {
			error ? console.log(error) : console.log(stdout)
		}))


	}

	async installZSHIntegrations() {

	}


	/**
	 * Uninstall the integration by deleting the workflow from the
	 * install path.
	 *
	 * @returns A promise when the uninstallation has completed
	 */
	async uninstall(): Promise<void> {
		exec(`rm -rf "${ INSTALL_PATH }"`, (error, stdout) => {
			error ? console.log(error) : console.log(stdout)
		})

	}
}
